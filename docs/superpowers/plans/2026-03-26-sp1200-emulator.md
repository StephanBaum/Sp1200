```js
// js/dsp/bitcrusher.js
const HALF_LEVELS = 2048; // 2^(12-1)

/**
 * Quantize a single float sample (-1 to 1) to 12-bit resolution.
 * Returns a float in [-1, 1) representing the quantized value.
 */
export function quantize12Bit(sample) {
  // Scale to 12-bit integer range [-2048, 2047]
  const scaled = Math.round(sample * HALF_LEVELS);
  // Clamp to 12-bit signed range
  const clamped = Math.max(-HALF_LEVELS, Math.min(HALF_LEVELS - 1, scaled));
  // Scale back to float
  return clamped / HALF_LEVELS;
}

/**
 * Quantize an entire Float32Array buffer to 12-bit.
 * Returns a new Float32Array.
 */
export function crushBuffer(buffer) {
  const output = new Float32Array(buffer.length);
  for (let i = 0; i < buffer.length; i++) {
    output[i] = quantize12Bit(buffer[i]);
  }
  return output;
}
```

```js
// js/dsp/resampler.js
import { SP_SAMPLE_RATE } from '../constants.js';

/**
 * Resample a buffer to 26.04kHz using linear interpolation (for input conversion).
 * This happens on sample load — not real-time critical.
 */
export function resampleTo26k(buffer, sourceSampleRate) {
  if (sourceSampleRate === SP_SAMPLE_RATE) {
    return new Float32Array(buffer);
  }

  const ratio = SP_SAMPLE_RATE / sourceSampleRate;
  const outputLength = Math.floor(buffer.length * ratio);
  const output = new Float32Array(outputLength);

  for (let i = 0; i < outputLength; i++) {
    const srcPos = i / ratio;
    const srcIndex = Math.floor(srcPos);
    const frac = srcPos - srcIndex;

    const s0 = buffer[srcIndex] || 0;
    const s1 = buffer[srcIndex + 1] || s0;
    output[i] = s0 + frac * (s1 - s0);
  }

  return output;
}

/**
 * Resample from SP rate to output rate using nearest-neighbor.
 * Nearest-neighbor is intentional — it produces the aliasing artifacts
 * that define the SP-1200 sound when pitching samples.
 */
export function resampleToOutput(buffer, outputSampleRate) {
  const ratio = outputSampleRate / SP_SAMPLE_RATE;
  const outputLength = Math.floor(buffer.length * ratio);
  const output = new Float32Array(outputLength);

  for (let i = 0; i < outputLength; i++) {
    const srcIndex = Math.floor(i / ratio);
    output[i] = buffer[Math.min(srcIndex, buffer.length - 1)];
  }

  return output;
}
```

```js
// js/dsp/filters.js
import { SP_SAMPLE_RATE } from '../constants.js';

/**
 * SSM2044-style 4-pole (24dB/oct) OTA ladder low-pass filter.
 * Used on channels 1-2 with pitch-tracking cutoff.
 *
 * Modeled as 4 cascaded one-pole sections with resonance feedback.
 */
export class SSM2044Filter {
  constructor(cutoff = 10000, resonance = 0) {
    this.cutoff = cutoff;
    this.resonance = resonance;
    this.sampleRate = SP_SAMPLE_RATE;
    // 4 cascaded stages
    this.s = [0, 0, 0, 0];
    this._updateCoefficients();
  }

  _updateCoefficients() {
    // Compute coefficient for one-pole: g = tan(pi * fc / sr)
    const fc = Math.max(20, Math.min(this.cutoff, this.sampleRate * 0.49));
    this.g = Math.tan(Math.PI * fc / this.sampleRate);
    this.gInv = 1 / (1 + this.g);
  }

  setCutoff(cutoff) {
    this.cutoff = cutoff;
    this._updateCoefficients();
  }

  setResonance(resonance) {
    this.resonance = Math.max(0, Math.min(resonance, 4));
  }

  process(input) {
    // Feedback from 4th stage (resonance)
    const feedback = this.resonance * this.s[3];
    let x = input - feedback;

    // Soft clip the feedback to prevent blowup (analog saturation)
    x = Math.tanh(x);

    // 4 cascaded one-pole low-pass filters
    for (let i = 0; i < 4; i++) {
      const v = (x - this.s[i]) * this.g * this.gInv;
      const y = v + this.s[i];
      this.s[i] = y + v;
      x = y;
    }

    return x;
  }

  reset() {
    this.s = [0, 0, 0, 0];
  }
}

/**
 * Simple fixed-frequency low-pass filter (2-pole).
 * Used on channels 3-6 for taming drum sounds.
 */
export class FixedFilter {
  constructor(cutoff = 8000) {
    this.sampleRate = SP_SAMPLE_RATE;
    this.s = [0, 0];
    const fc = Math.max(20, Math.min(cutoff, this.sampleRate * 0.49));
    this.g = Math.tan(Math.PI * fc / this.sampleRate);
    this.gInv = 1 / (1 + this.g);
  }

  process(input) {
    let x = input;
    for (let i = 0; i < 2; i++) {
      const v = (x - this.s[i]) * this.g * this.gInv;
      const y = v + this.s[i];
      this.s[i] = y + v;
      x = y;
    }
    return x;
  }

  reset() {
    this.s = [0, 0];
  }
}
```

```js
// js/dsp/voice.js

/**
 * Single voice/channel of the SP-1200.
 * Handles sample playback with pitch, velocity, decay, reverse, and loop.
 * Uses nearest-neighbor interpolation for authentic aliasing.
 */
export class Voice {
  constructor(channelIndex) {
    this.channelIndex = channelIndex;
    this.sample = null;
    this.active = false;
    this.position = 0;
    this.velocity = 0;
    this.pitch = 1.0;
    this.decayRate = 1.0;    // 1.0 = no decay, 0 = instant
    this.decayLevel = 1.0;
    this.reversed = false;
    this.loopEnabled = false;
    this.loopStart = 0;
    this.loopEnd = 0;
    this.startPoint = 0;
    this.endPoint = 0;
  }

  loadSample(buffer) {
    this.sample = buffer;
    this.startPoint = 0;
    this.endPoint = buffer.length - 1;
    this.loopEnd = buffer.length - 1;
  }

  trigger(velocity) {
    if (!this.sample) return;
    this.active = true;
    this.velocity = velocity / 127;
    this.decayLevel = 1.0;

    if (this.reversed) {
      this.position = this.endPoint;
    } else {
      this.position = this.startPoint;
    }
  }

  stop() {
    this.active = false;
    this.position = 0;
  }

  setPitch(rate) {
    this.pitch = rate;
  }

  setDecay(amount) {
    // amount: 0 = instant silence, 1 = no decay (sustain)
    // Convert to per-sample multiplier
    this.decayRate = 0.9999 + (amount * 0.0001);
    // For shorter decay, use exponential curve
    if (amount < 1) {
      this.decayRate = 0.995 + (amount * 0.005);
    }
  }

  setReversed(reversed) {
    this.reversed = reversed;
  }

  setLoop(enabled, start = 0, end = 0) {
    this.loopEnabled = enabled;
    if (enabled) {
      this.loopStart = start;
      this.loopEnd = end;
    }
  }

  setTruncate(start, end) {
    this.startPoint = start;
    this.endPoint = end;
  }

  process() {
    if (!this.active || !this.sample) return 0;

    // Nearest-neighbor: grab sample at integer position
    const index = Math.floor(this.position);

    // Bounds check
    if (this.reversed) {
      if (index < this.startPoint) {
        if (this.loopEnabled) {
          this.position = this.loopEnd;
        } else {
          this.active = false;
          return 0;
        }
      }
    } else {
      if (index > this.endPoint) {
        if (this.loopEnabled) {
          this.position = this.loopStart;
        } else {
          this.active = false;
          return 0;
        }
      }
    }

    const safeIndex = Math.max(0, Math.min(index, this.sample.length - 1));
    const raw = this.sample[safeIndex];

    // Apply velocity and decay
    const out = raw * this.velocity * this.decayLevel;

    // Advance position
    if (this.reversed) {
      this.position -= this.pitch;
    } else {
      this.position += this.pitch;
    }

    // Apply decay
    this.decayLevel *= this.decayRate;
    if (this.decayLevel < 0.001) {
      this.active = false;
    }

    return out;
  }
}
```

```js
// js/dsp/mixer.js
import { NUM_PADS } from '../constants.js';

/**
 * 8-channel mixer. Each channel has volume and pan.
 * Outputs stereo (left, right).
 * Uses constant-power panning (sin/cos law).
 */
export class Mixer {
  constructor() {
    this.channels = Array.from({ length: NUM_PADS }, () => ({
      volume: 1.0,
      pan: 0,       // -1 (left) to 1 (right)
      gainL: Math.SQRT1_2,
      gainR: Math.SQRT1_2,
    }));
  }

  setVolume(channel, volume) {
    this.channels[channel].volume = Math.max(0, Math.min(1, volume));
  }

  setPan(channel, pan) {
    const ch = this.channels[channel];
    ch.pan = Math.max(-1, Math.min(1, pan));
    // Constant-power pan law
    const angle = (ch.pan + 1) * Math.PI / 4; // 0 to PI/2
    ch.gainL = Math.cos(angle);
    ch.gainR = Math.sin(angle);
  }

  /**
   * Process 8 mono input samples into stereo output.
   * @param {Float32Array} inputs - 8 mono samples, one per channel
   * @returns {[number, number]} - [left, right]
   */
  process(inputs) {
    let left = 0;
    let right = 0;

    for (let i = 0; i < NUM_PADS; i++) {
      const signal = inputs[i] * this.channels[i].volume;
      left += signal * this.channels[i].gainL;
      right += signal * this.channels[i].gainR;
    }

    return [left, right];
  }
}
```

```js
// js/sequencer/clock.js
import { PPQN } from '../constants.js';

export class Clock {
  constructor(sampleRate) {
    this.sampleRate = sampleRate;
    this.bpm = 90;
    this.samplesPerTick = 0;
    this.sampleCounter = 0;
    this.tick = 0;
    this.playing = false;
    this._calcSamplesPerTick();
  }

  _calcSamplesPerTick() {
    // BPM = quarter notes per minute
    // Samples per quarter note = sampleRate * 60 / BPM
    // Samples per tick = samples per quarter / PPQN
    this.samplesPerTick = (this.sampleRate * 60) / (this.bpm * PPQN);
  }

  setBpm(bpm) {
    this.bpm = Math.max(30, Math.min(250, bpm));
    this._calcSamplesPerTick();
  }

  start() {
    this.playing = true;
    this.tick = 0;
    this.sampleCounter = 0;
  }

  stop() {
    this.playing = false;
    this.tick = 0;
    this.sampleCounter = 0;
  }

  /**
   * Call once per audio sample. Returns the tick number if a tick
   * boundary was crossed, or null otherwise.
   */
  advance() {
    if (!this.playing) return null;

    this.sampleCounter++;
    const nextTickAt = Math.floor((this.tick + 1) * this.samplesPerTick);

    if (this.sampleCounter >= nextTickAt) {
      this.tick++;
      return this.tick;
    }

    return null;
  }

  /**
   * Convert a tick number to bar/beat/sixteenth position.
   */
  getPosition(tick) {
    const ticksPerBeat = PPQN;
    const ticksPerBar = PPQN * 4;
    const ticksPer16th = PPQN / 4;

    const bar = Math.floor(tick / ticksPerBar);
    const beatTick = tick % ticksPerBar;
    const beat = Math.floor(beatTick / ticksPerBeat);
    const sixteenthTick = beatTick % ticksPerBeat;
    const sixteenth = Math.floor(sixteenthTick / ticksPer16th);

    return { bar, beat, sixteenth, tick };
  }
}
```

```js
// js/sequencer/swing.js
import { PPQN } from '../constants.js';

const TICKS_PER_16TH = PPQN / 4; // 24

/**
 * Get the swing delay offset for a given tick position.
 * Swing delays every even-numbered 16th note (the "e" and "a" of each beat pair).
 *
 * @param {number} tick - Tick position within the pattern
 * @param {number} swingPercent - 50 (straight) to 75 (max swing)
 * @returns {number} - Tick offset to add (0 for on-beat 16ths)
 */
export function getSwingOffset(tick, swingPercent) {
  if (swingPercent <= 50) return 0;

  // Which 16th note is this?
  const sixteenthIndex = Math.floor(tick / TICKS_PER_16TH);
  const posInSixteenth = tick % TICKS_PER_16TH;

  // Only delay odd-indexed 16ths (the "off" 16th in each pair)
  if (sixteenthIndex % 2 === 0 || posInSixteenth !== 0) return 0;

  // Swing amount: 50% = no delay, 75% = delay by half a 16th note
  const swingAmount = (swingPercent - 50) / 25; // 0 to 1
  return Math.round(TICKS_PER_16TH * swingAmount * 0.5);
}

/**
 * Apply swing to a tick position.
 * @returns {number} Adjusted tick position
 */
export function applySwing(tick, swingPercent) {
  return tick + getSwingOffset(tick, swingPercent);
}
```

```js
import { PPQN, NUM_PADS } from '../constants.js';

export class PatternEvent {
  constructor(tick, velocity = 127, pitchOffset = 0) {
    this.tick = tick;
    this.velocity = velocity;
    this.pitchOffset = pitchOffset;
  }
}

class Track {
  constructor() {
    this.events = [];
  }

  addEvent(event) {
    this.events.push(event);
    this.events.sort((a, b) => a.tick - b.tick);
  }

  removeEventAtTick(tick) {
    const idx = this.events.findIndex(e => e.tick === tick);
    if (idx !== -1) this.events.splice(idx, 1);
  }

  clear() {
    this.events = [];
  }

  getEventsAtTick(tick) {
    return this.events.filter(e => e.tick === tick);
  }
}

export class Pattern {
  constructor() {
    this.bars = 1;
    this.totalTicks = PPQN * 4; // 4 beats per bar
    this.tracks = Array.from({ length: NUM_PADS }, () => new Track());
  }

  setBars(bars) {
    this.bars = Math.max(1, Math.min(4, bars));
    this.totalTicks = PPQN * 4 * this.bars;
  }

  addEvent(trackIndex, event) {
    if (trackIndex >= 0 && trackIndex < NUM_PADS) {
      this.tracks[trackIndex].addEvent(event);
    }
  }

  removeEvent(trackIndex, tick) {
    if (trackIndex >= 0 && trackIndex < NUM_PADS) {
      this.tracks[trackIndex].removeEventAtTick(tick);
    }
  }

  getEventsAtTick(tick) {
    const results = [];
    for (let i = 0; i < NUM_PADS; i++) {
      const events = this.tracks[i].getEventsAtTick(tick);
      for (const e of events) {
        results.push({ track: i, ...e });
      }
    }
    return results;
  }

  quantizeTick(tick, gridSize) {
    return Math.round(tick / gridSize) * gridSize;
  }

  clearTrack(trackIndex) {
    if (trackIndex >= 0 && trackIndex < NUM_PADS) {
      this.tracks[trackIndex].clear();
    }
  }

  clear() {
    for (const track of this.tracks) track.clear();
  }

  serialize() {
    return {
      bars: this.bars,
      tracks: this.tracks.map(t => ({
        events: t.events.map(e => ({
          tick: e.tick,
          velocity: e.velocity,
          pitchOffset: e.pitchOffset,
        })),
      })),
    };
  }

  static deserialize(data) {
    const p = new Pattern();
    p.setBars(data.bars);
    data.tracks.forEach((trackData, i) => {
      for (const e of trackData.events) {
        p.addEvent(i, new PatternEvent(e.tick, e.velocity, e.pitchOffset));
      }
    });
    return p;
  }
}
```

```js
// js/sequencer/song.js
import { MAX_SONG_ENTRIES } from '../constants.js';

export class SongEntry {
  constructor(pattern, repeats = 1) {
    this.pattern = pattern;
    this.repeats = Math.max(1, repeats);
  }
}

export class Song {
  constructor() {
    this.entries = [];
    this.currentIndex = 0;
    this.currentRepeat = 0;
  }

  addEntry(entry) {
    if (this.entries.length < MAX_SONG_ENTRIES) {
      this.entries.push(entry);
    }
  }

  removeEntry(index) {
    if (index >= 0 && index < this.entries.length) {
      this.entries.splice(index, 1);
    }
  }

  start() {
    this.currentIndex = 0;
    this.currentRepeat = 0;
  }

  currentPattern() {
    if (this.isFinished()) return -1;
    return this.entries[this.currentIndex].pattern;
  }

  advanceRepeat() {
    if (this.isFinished()) return;

    this.currentRepeat++;
    if (this.currentRepeat >= this.entries[this.currentIndex].repeats) {
      this.currentIndex++;
      this.currentRepeat = 0;
    }
  }

  isFinished() {
    return this.currentIndex >= this.entries.length;
  }

  getPosition() {
    return {
      entryIndex: this.currentIndex,
      repeat: this.currentRepeat,
      pattern: this.currentPattern(),
    };
  }

  serialize() {
    return {
      entries: this.entries.map(e => ({
        pattern: e.pattern,
        repeats: e.repeats,
      })),
    };
  }

  static deserialize(data) {
    const s = new Song();
    for (const e of data.entries) {
      s.addEntry(new SongEntry(e.pattern, e.repeats));
    }
    return s;
  }
}
```

```js
// js/audio/sp1200-processor.js
import { Voice } from '../dsp/voice.js';
import { SSM2044Filter, FixedFilter } from '../dsp/filters.js';
import { Mixer } from '../dsp/mixer.js';
import { Clock } from '../sequencer/clock.js';
import { getSwingOffset } from '../sequencer/swing.js';
import { Pattern, PatternEvent } from '../sequencer/pattern.js';
import { Song } from '../sequencer/song.js';
import {
  NUM_PADS, PPQN, MAX_PATTERNS, BPM_DEFAULT,
  FILTER_DYNAMIC, FILTER_FIXED, FILTER_NONE,
} from '../constants.js';

class SP1200Processor extends AudioWorkletProcessor {
  constructor() {
    super();

    this.voices = Array.from({ length: NUM_PADS }, (_, i) => new Voice(i));
    this.filters = this._createFilters();
    this.mixer = new Mixer();
    this.clock = new Clock(sampleRate);
    this.clock.setBpm(BPM_DEFAULT);

    this.patterns = Array.from({ length: MAX_PATTERNS }, () => new Pattern());
    this.currentPatternIndex = 0;
    this.song = new Song();

    this.mode = 'pattern'; // 'pattern' | 'song'
    this.recording = false;
    this.swing = 50;
    this.quantizeGrid = PPQN / 4; // 1/16th default

    this.autoRepeatPads = new Set();
    this.autoRepeatCounter = 0;

    this.port.onmessage = (e) => this._handleMessage(e.data);
  }

  _createFilters() {
    const filters = [];
    for (let i = 0; i < NUM_PADS; i++) {
      if (FILTER_DYNAMIC.includes(i)) {
        filters.push(new SSM2044Filter(10000, 0));
      } else if (FILTER_FIXED.includes(i)) {
        filters.push(new FixedFilter(8000));
      } else {
        filters.push(null); // FILTER_NONE channels — no filter
      }
    }
    return filters;
  }

  _handleMessage(msg) {
    switch (msg.type) {
      case 'trigger':
        this.voices[msg.pad].trigger(msg.velocity);
        break;

      case 'stop-voice':
        this.voices[msg.pad].stop();
        break;

      case 'load-sample':
        this.voices[msg.pad].loadSample(new Float32Array(msg.buffer));
        break;

      case 'set-param':
        this._setParam(msg.param, msg.pad, msg.value);
        break;

      case 'transport':
        if (msg.action === 'play') {
          this.clock.start();
          if (this.mode === 'song') this.song.start();
        } else if (msg.action === 'stop') {
          this.clock.stop();
          this.recording = false;
        } else if (msg.action === 'record') {
          this.recording = true;
          this.clock.start();
        }
        break;

      case 'set-bpm':
        this.clock.setBpm(msg.bpm);
        break;

      case 'set-swing':
        this.swing = msg.amount;
        break;

      case 'set-quantize':
        this.quantizeGrid = msg.grid;
        break;

      case 'pattern-select':
        this.currentPatternIndex = Math.max(0, Math.min(MAX_PATTERNS - 1, msg.number));
        break;

      case 'step-edit': {
        const pat = this.patterns[this.currentPatternIndex];
        if (msg.event) {
          pat.addEvent(msg.track, new PatternEvent(msg.step, msg.event.velocity, msg.event.pitchOffset));
        } else {
          pat.removeEvent(msg.track, msg.step);
        }
        break;
      }

      case 'song-chain':
        this.song = Song.deserialize(msg.entries);
        break;

      case 'set-mode':
        this.mode = msg.mode;
        break;

      case 'auto-repeat':
        if (msg.enabled) {
          this.autoRepeatPads.add(msg.pad);
        } else {
          this.autoRepeatPads.delete(msg.pad);
        }
        break;
    }
  }

  _setParam(param, pad, value) {
    switch (param) {
      case 'pitch':
        this.voices[pad].setPitch(value);
        // Dynamic filter channels: cutoff tracks pitch
        if (FILTER_DYNAMIC.includes(pad) && this.filters[pad]) {
          this.filters[pad].setCutoff(10000 * value);
        }
        break;
      case 'volume':
        this.mixer.setVolume(pad, value);
        break;
      case 'pan':
        this.mixer.setPan(pad, value);
        break;
      case 'decay':
        this.voices[pad].setDecay(value);
        break;
      case 'reverse':
        this.voices[pad].setReversed(value);
        break;
      case 'loop':
        this.voices[pad].setLoop(value.enabled, value.start, value.end);
        break;
      case 'truncate':
        this.voices[pad].setTruncate(value.start, value.end);
        break;
    }
  }

  _getCurrentPattern() {
    if (this.mode === 'song' && !this.song.isFinished()) {
      const idx = this.song.currentPattern();
      return idx >= 0 ? this.patterns[idx] : this.patterns[this.currentPatternIndex];
    }
    return this.patterns[this.currentPatternIndex];
  }

  _processSequencer() {
    const tickResult = this.clock.advance();
    if (tickResult === null) return;

    const pattern = this._getCurrentPattern();
    const patternTick = tickResult % pattern.totalTicks;

    // Check if pattern looped
    if (patternTick === 0 && tickResult > 0) {
      if (this.mode === 'song') {
        this.song.advanceRepeat();
        if (this.song.isFinished()) {
          this.clock.stop();
          this.port.postMessage({ type: 'song-end' });
          return;
        }
        this.port.postMessage({ type: 'song-position', ...this.song.getPosition() });
      }
      this.port.postMessage({ type: 'pattern-end' });
    }

    // Apply swing to get the adjusted tick
    const swingOffset = getSwingOffset(patternTick, this.swing);
    const checkTick = patternTick - swingOffset; // check non-swung position

    // Fire events at this tick
    const events = pattern.getEventsAtTick(checkTick >= 0 ? checkTick : patternTick);
    for (const evt of events) {
      this.voices[evt.track].trigger(evt.velocity);
      this.port.postMessage({ type: 'trigger-visual', pad: evt.track });
    }

    // Auto-repeat
    if (patternTick % this.quantizeGrid === 0) {
      for (const pad of this.autoRepeatPads) {
        this.voices[pad].trigger(100);
        this.port.postMessage({ type: 'trigger-visual', pad });
      }
    }

    // Send tick for display
    const pos = this.clock.getPosition(tickResult);
    this.port.postMessage({ type: 'tick', ...pos });
  }

  process(inputs, outputs) {
    const output = outputs[0];
    const left = output[0];
    const right = output[1];

    for (let s = 0; s < left.length; s++) {
      this._processSequencer();

      // Process all 8 voices through filters
      const channelSamples = new Float32Array(NUM_PADS);
      for (let ch = 0; ch < NUM_PADS; ch++) {
        let sample = this.voices[ch].process();

        // Apply channel filter
        if (this.filters[ch]) {
          sample = this.filters[ch].process(sample);
        }

        channelSamples[ch] = sample;
      }

      // Mix to stereo
      const [l, r] = this.mixer.process(channelSamples);
      left[s] = l;
      right[s] = r;
    }

    return true;
  }
}

registerProcessor('sp1200-processor', SP1200Processor);
```

```js
// js/audio/sample-loader.js
import { resampleTo26k } from '../dsp/resampler.js';
import { crushBuffer } from '../dsp/bitcrusher.js';
import { SP_SAMPLE_RATE, BANK_SAMPLE_FRAMES, NUM_PADS } from '../constants.js';

/**
 * Tracks sample memory usage per bank.
 * Banks A-D each hold ~2.5 seconds at 26.04kHz.
 */
export class SampleMemory {
  constructor() {
    // frames used per bank
    this.bankUsage = [0, 0, 0, 0];
  }

  getBank(padIndex) {
    return Math.floor(padIndex / NUM_PADS);
  }

  getRemainingFrames(bank) {
    return BANK_SAMPLE_FRAMES - this.bankUsage[bank];
  }

  getRemainingSeconds(bank) {
    return this.getRemainingFrames(bank) / SP_SAMPLE_RATE;
  }

  canFit(bank, frames) {
    return this.getRemainingFrames(bank) >= frames;
  }

  allocate(bank, frames) {
    this.bankUsage[bank] += frames;
  }

  free(bank, frames) {
    this.bankUsage[bank] = Math.max(0, this.bankUsage[bank] - frames);
  }
}

/**
 * Decode an audio file, resample to 26.04kHz, and quantize to 12-bit.
 *
 * @param {AudioContext} audioContext - For decoding
 * @param {ArrayBuffer} fileData - Raw file bytes (WAV, MP3, etc.)
 * @returns {Promise<Float32Array>} - Processed mono sample at 26.04kHz, 12-bit
 */
export async function loadSampleFromFile(audioContext, fileData) {
  const audioBuffer = await audioContext.decodeAudioData(fileData);

  // Mix to mono if stereo
  let mono;
  if (audioBuffer.numberOfChannels === 1) {
    mono = audioBuffer.getChannelData(0);
  } else {
    const left = audioBuffer.getChannelData(0);
    const right = audioBuffer.getChannelData(1);
    mono = new Float32Array(left.length);
    for (let i = 0; i < left.length; i++) {
      mono[i] = (left[i] + right[i]) * 0.5;
    }
  }

  // Resample to 26.04 kHz
  const resampled = resampleTo26k(mono, audioBuffer.sampleRate);

  // Quantize to 12-bit
  const crushed = crushBuffer(resampled);

  return crushed;
}

/**
 * Process raw PCM from microphone recording.
 * Input is already at audioContext.sampleRate.
 */
export function processMicCapture(rawBuffer, sourceSampleRate) {
  const resampled = resampleTo26k(rawBuffer, sourceSampleRate);
  return crushBuffer(resampled);
}
```

```js
// js/ui/pads.js

const PAD_KEYS = ['z', 'x', 'c', 'v', 'a', 's', 'd', 'f'];

export class PadsUI {
  constructor(engine) {
    this.engine = engine;
    this.padElements = document.querySelectorAll('.pad');
    this.currentBank = 0;
    this._bindMouse();
    this._bindKeyboard();
  }

  _bindMouse() {
    this.padElements.forEach((el) => {
      el.addEventListener('mousedown', (e) => {
        const pad = parseInt(el.dataset.pad, 10);
        const velocity = this._velocityFromClick(e, el);
        this.engine.trigger(pad, velocity);
        this._flash(pad);
      });
    });
  }

  _bindKeyboard() {
    document.addEventListener('keydown', (e) => {
      if (e.repeat) return;
      const padIndex = PAD_KEYS.indexOf(e.key.toLowerCase());
      if (padIndex === -1) return;
      this.engine.trigger(padIndex, 100);
      this._flash(padIndex);
    });
  }

  /**
   * Velocity from click position: center = 127, edge = 40.
   */
  _velocityFromClick(event, element) {
    const rect = element.getBoundingClientRect();
    const centerX = rect.width / 2;
    const centerY = rect.height / 2;
    const dx = event.clientX - rect.left - centerX;
    const dy = event.clientY - rect.top - centerY;
    const dist = Math.sqrt(dx * dx + dy * dy);
    const maxDist = Math.sqrt(centerX * centerX + centerY * centerY);
    const normalized = 1 - Math.min(dist / maxDist, 1);
    return Math.round(40 + normalized * 87); // 40-127
  }

  _flash(padIndex) {
    const el = this.padElements[padIndex];
    if (!el) return;
    el.classList.add('triggered');
    setTimeout(() => el.classList.remove('triggered'), 100);
  }

  flashPad(padIndex) {
    this._flash(padIndex);
  }

  setBank(bank) {
    this.currentBank = bank;
  }
}
```

```js
// js/ui/faders.js

export class FadersUI {
  constructor(engine) {
    this.engine = engine;
    this.mode = 'volume'; // 'volume' | 'pitch'
    this.values = new Float32Array(8).fill(0.75); // default 75%
    this.thumbs = document.querySelectorAll('.fader-thumb');
    this.tracks = document.querySelectorAll('.fader-track');
    this._bindModeSwitcher();
    this._bindDrag();
    this._updateAllPositions();
  }

  _bindModeSwitcher() {
    const buttons = document.querySelectorAll('.fader-mode-switch .mode-btn');
    buttons.forEach(btn => {
      btn.addEventListener('click', () => {
        buttons.forEach(b => b.classList.remove('active'));
        btn.classList.add('active');
        this.mode = btn.dataset.mode;
      });
    });
  }

  _bindDrag() {
    this.thumbs.forEach((thumb, index) => {
      let dragging = false;
      const track = this.tracks[index];

      thumb.addEventListener('mousedown', (e) => {
        dragging = true;
        e.preventDefault();
      });

      document.addEventListener('mousemove', (e) => {
        if (!dragging) return;
        const rect = track.getBoundingClientRect();
        // Inverted: top = 1, bottom = 0
        const y = 1 - Math.max(0, Math.min(1, (e.clientY - rect.top) / rect.height));
        this.values[index] = y;
        this._updatePosition(index);
        this._sendValue(index);
      });

      document.addEventListener('mouseup', () => {
        dragging = false;
      });
    });
  }

  _updatePosition(index) {
    const thumb = this.thumbs[index];
    const track = this.tracks[index];
    if (!thumb || !track) return;
    // Position: 0 = bottom, 1 = top
    const percent = (1 - this.values[index]) * 100;
    thumb.style.top = `${percent}%`;
  }

  _updateAllPositions() {
    for (let i = 0; i < 8; i++) this._updatePosition(i);
  }

  _sendValue(index) {
    if (this.mode === 'volume') {
      this.engine.setParam('volume', index, this.values[index]);
    } else {
      // Pitch: center (0.5) = 1.0x, top (1.0) = 2.0x, bottom (0.0) = 0.5x
      const pitch = 0.5 + this.values[index] * 1.5;
      this.engine.setParam('pitch', index, pitch);
    }
  }
}
```

```js
// js/storage/indexeddb.js

const DB_NAME = 'sp1200';
const DB_VERSION = 1;
const STORE_NAME = 'projects';

function openDB() {
  return new Promise((resolve, reject) => {
    const req = indexedDB.open(DB_NAME, DB_VERSION);
    req.onupgradeneeded = () => {
      const db = req.result;
      if (!db.objectStoreNames.contains(STORE_NAME)) {
        db.createObjectStore(STORE_NAME, { keyPath: 'name' });
      }
    };
    req.onsuccess = () => resolve(req.result);
    req.onerror = () => reject(req.error);
  });
}

export class ProjectStore {
  static serialize(project) {
    return JSON.stringify(project);
  }

  static deserialize(json) {
    return JSON.parse(json);
  }

  static async save(project) {
    const db = await openDB();
    return new Promise((resolve, reject) => {
      const tx = db.transaction(STORE_NAME, 'readwrite');
      const store = tx.objectStore(STORE_NAME);
      store.put(project);
      tx.oncomplete = () => resolve();
      tx.onerror = () => reject(tx.error);
    });
  }

  static async load(name) {
    const db = await openDB();
    return new Promise((resolve, reject) => {
      const tx = db.transaction(STORE_NAME, 'readonly');
      const store = tx.objectStore(STORE_NAME);
      const req = store.get(name);
      req.onsuccess = () => resolve(req.result || null);
      req.onerror = () => reject(req.error);
    });
  }

  static async list() {
    const db = await openDB();
    return new Promise((resolve, reject) => {
      const tx = db.transaction(STORE_NAME, 'readonly');
      const store = tx.objectStore(STORE_NAME);
      const req = store.getAllKeys();
      req.onsuccess = () => resolve(req.result);
      req.onerror = () => reject(req.error);
    });
  }

  static async remove(name) {
    const db = await openDB();
    return new Promise((resolve, reject) => {
      const tx = db.transaction(STORE_NAME, 'readwrite');
      const store = tx.objectStore(STORE_NAME);
      store.delete(name);
      tx.oncomplete = () => resolve();
      tx.onerror = () => reject(tx.error);
    });
  }
}
```

```js
// js/storage/export.js

const CURRENT_VERSION = 1;

export function exportProject(project) {
  const exportData = {
    version: CURRENT_VERSION,
    name: project.name,
    bpm: project.bpm,
    swing: project.swing,
    patterns: project.patterns,
    song: project.song,
    samples: {},
  };

  // Encode sample data
  for (const [pad, sample] of Object.entries(project.samples || {})) {
    exportData.samples[pad] = {
      name: sample.name,
      data: Array.from(sample.data),
    };
  }

  return JSON.stringify(exportData, null, 2);
}

export function importProject(jsonString) {
  const data = JSON.parse(jsonString);

  if (data.version !== CURRENT_VERSION) {
    throw new Error(`Unsupported version: ${data.version}`);
  }

  return {
    name: data.name,
    bpm: data.bpm,
    swing: data.swing,
    patterns: data.patterns || [],
    song: data.song || { entries: [] },
    samples: data.samples || {},
  };
}

/**
 * Trigger a browser download of the project file.
 */
export function downloadProject(project) {
  const json = exportProject(project);
  const blob = new Blob([json], { type: 'application/json' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `${project.name || 'sp1200-project'}.sp12.json`;
  a.click();
  URL.revokeObjectURL(url);
}
```
